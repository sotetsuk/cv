\newcommand{\EducationEntry}[4]{
		\noindent \textbf{#1} \hfill      % Study

		\noindent\hangindent=2em\hangafter=0 \small #4 % Description
		\normalsize \par}

		\colorbox{Black}{\color{White}#2} \par  % Duration
		\noindent \textit{#3} \par              % Company
		\noindent\hangindent=2em\hangafter=0 \small #4 % Description
		\normalsize \par}
